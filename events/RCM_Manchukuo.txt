# State religion
country_event = {
	id = Manchukuo.3
	title = Manchukuo.3.t
	picture = GFX_report_event_generic_read_write
	desc = Manchukuo.3.d
	
	is_triggered_only = yes
	
		option = {
				name = Manchukuo.3.a
					if = {
				limit = {
				OR = {
					has_autonomy_state = autonomy_colony
					has_autonomy_state = autonomy_puppet
					has_autonomy_state = autonomy_integrated_puppet
					}
				}
				add_ideas = state_shintoism
				add_stability = -0.05
					add_autonomy_ratio = {
					value = -0.05
					localization = MAN_Improve_autonomy
					}
				}
				
					if = {
				limit = {
				OR = {
					has_autonomy_state = autonomy_dominion
					is_puppet = no
					}
				}
				add_ideas = Manchukuo_state_shintoism
				add_stability = 0.05
					add_autonomy_ratio = {
					value = 0.05
					localization = MAN_Improve_autonomy
					}
				}
			}
}

	is_triggered_only = yes
	
		option = {

				if = {

                        add_to_faction = MAN
						}

			}
		option = {
				name = Manchukuo.4.b
				ai_chance = { factor = 10 }
					add_stability = 0.2
			}
}
