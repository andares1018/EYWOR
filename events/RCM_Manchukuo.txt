	desc = EFM_Manchukuo.9.d
	
	is_triggered_only = yes
	
	option = {
		name = EFM_Manchukuo.9.a
		effect_tooltip = {
			if = {
				limit = {
					has_dlc = "Together for Victory"
				}
				MAN = {
					add_autonomy_ratio = {
						value = -0.05
					}
					add_ideas = integration_movement
				}
				else = {
					set_autonomy = {
						target = MAN
						autonomous_state = autonomy_integrated_puppet
					}
				}
			}
		}
	}
}
