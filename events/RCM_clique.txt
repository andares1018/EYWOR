country_event = {
	id = clique.3
	hidden = yes
	trigger = {
		is_ai = yes
		is_warlord = yes
		has_equipment = {
			infantry_equipment > 199
		}
		has_war_with = JAP 
		NOT = {
			any_neighbor_country = {
				has_war_with = ROOT
			}
		}
		any_other_country = {
			is_legit_china = yes
			is_ai = yes
			has_war_together_with = ROOT
			is_faction_leader = yes
			is_in_faction_with = ROOT
			has_war_with = JAP
		}
	}
	mean_time_to_happen = {
		days = 7
	}
	immediate = {
		random_other_country = {
			limit = {
				is_legit_china = yes
				is_ai = yes
				has_war_together_with = ROOT
				is_faction_leader = yes
				is_in_faction_with = ROOT
				has_war_with = JAP
			}
			ROOT = {
				send_equipment = {
					target = PREV
					type = infantry_equipment
					amount = 200
				}
			}
		}
	}
	option = {
	}
